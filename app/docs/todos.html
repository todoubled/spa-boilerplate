	localStorage: new Store 'todos'
	
	getDone: (todo) ->
		todo.get 'done'

	done: ->
		@filter @getDone
 
	remaining: ->
		@without.apply this, @done()

	nextOrder: ->
		return 1 if !@length 
		return @last().get('order') + 1

	comparator: (todo) -> 
		todo.get 'order'
